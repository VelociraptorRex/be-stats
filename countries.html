Установи его командой: pip install openpyxl") from exc # Коды для flagcdn.com. При необходимости можно спокойно дополнять словарь. FLAG_CODES = { "Австралия": "au", "Австрия": "at", "Алжир": "dz", "Англия": "gb-eng", "Аргентина": "ar", "Бельгия": "be", "Босния и Герцеговина": "ba", "Бразилия": "br", "Гаити": "ht", "Германия": "de", "ДР Конго": "cd", "Египет": "eg", "Иран": "ir", "Ирак": "iq", "Испания": "es", "Италия": "it", "Кабо-Верде": "cv", "Канада": "ca", "Катар": "qa", "Кот д'Ивуар": "ci", "Коста-Рика": "cr", "Кюрасао": "cw", "Марокко": "ma", "Мексика": "mx", "Нидерланды": "nl", "Новая Зеландия": "nz", "Норвегия": "no", "Парагвай": "py", "Португалия": "pt", "Саудовская Аравия": "sa", "Сенегал": "sn", "Сербия": "rs", "США": "us", "Тунис": "tn", "Турция": "tr", "Уругвай": "uy", "Франция": "fr", "Хорватия": "hr", "Чехия": "cz", "Швейцария": "ch", "Швеция": "se", "Шотландия": "gb-sct", "Эквадор": "ec", "Южная Корея": "kr", "ЮАР": "za", "Япония": "jp", } MATCH_DASH = "—" def clean_match_name(value): """Берём только строку с командами, отбрасывая счёт после переноса строки.""" if not isinstance(value, str): return None value = value.strip() if MATCH_DASH not in value: return None return value.split("\n", 1)[0].strip() def split_countries(match_name): left, right = match_name.split(MATCH_DASH, 1) return left.strip(), right.strip() def parse_bank(value): if isinstance(value, (int, float)): return float(value) return None def read_match_blocks(xlsx_path, sheet_name="Журнал"): wb = load_workbook(xlsx_path, data_only=True) if sheet_name not in wb.sheetnames: raise SystemExit(f"В Excel-файле нет листа {sheet_name!r}. Найдены листы: {', '.join(wb.sheetnames)}") ws = wb[sheet_name] blocks = [] for row in range(1, ws.max_row + 1): match_name = clean_match_name(ws.cell(row, 1).value) if not match_name: continue # Структура твоего журнала: # строка матча = ставка/банк Gerzog; # строка + 2 = ставка/банк Yukon. gerzog_bank = parse_bank(ws.cell(row, 4).value) yukon_bank = parse_bank(ws.cell(row + 2, 4).value) if row + 2 <= ws.max_row else None blocks.append({ "row": row, "match": match_name, "gerzog_bank": gerzog_bank, "yukon_bank": yukon_bank, }) return blocks def country_stats(blocks, player_key, initial_bank=1000.0): """ Считаем профит по изменению банка. Журнал идёт сверху вниз от новых матчей к старым, поэтому считаем снизу вверх. Профит каждой ставки начисляется обеим странам матча. """ previous_bank = float(initial_bank) stats = defaultdict(lambda: {"total": 0.0, "count": 0, "wins": 0, "max_win": 0.0}) for block in reversed(blocks): bank = block[player_key] if bank is None: continue profit = bank - previous_bank previous_bank = bank country_a, country_b = split_countries(block["match"]) for country in (country_a, country_b): stats[country]["total"] += profit stats[country]["count"] += 1 if profit > 0: stats[country]["wins"] += 1 stats[country]["max_win"] = max(stats[country]["max_win"], profit) rows = [] for country, values in stats.items(): count = values["count"] total = values["total"] avg = total / count if count else 0 rows.append({ "country": country, "total": total, "count": count, "wins": values["wins"], "avg": avg, "max_win": values["max_win"], }) rows.sort(key=lambda x: x["country"], reverse=True) rows.sort(key=lambda x: x["total"], reverse=True) return rows def fmt_total(value): value = round(value, 1) return str(int(value)) if value.is_integer() else str(value).replace(".", ",") def fmt_avg(value): return f"{value:.1f}".replace(".", ",") def sort_number(value): return f"{float(value):.1f}" def row_html(item): country = item["country"] total = item["total"] count = item["count"] wins = item["wins"] avg = item["avg"] max_win = item["max_win"] code = FLAG_CODES.get(country) safe_country = html.escape(country, quote=True) name_class = "country-name-long" if len(country) >= 18 else "" if code: flag = f'' else: # Если страна новая и её нет в FLAG_CODES, таблица не ломается: просто будет без картинки. flag = '' return ( f'' f'{flag}{safe_country}' f'{fmt_total(total)}' f'{wins}/{count}' f'{fmt_avg(avg)}' f'{fmt_max_win(max_win)}' ) def build_rows(rows, limit=12): return "".join(row_html(item) for item in rows[:limit]) def main(): parser = argparse.ArgumentParser(description="Пересобрать countries.html из Excel, сохранив старое оформление.") parser.add_argument("--xlsx", default="Ставки на ЧМ-2026-20-42-24-06-2026.xlsx", help="Путь к Excel-файлу") parser.add_argument("--template", default="countries_template.html", help="HTML-шаблон со специальными метками") parser.add_argument("--out", default="/Users/yukon/Documents/Gevalex/SiteGA/betting-stats-main 3/ready/countries_EKS3.html", help="Куда сохранить готовый HTML") parser.add_argument("--limit", type=int, default=12, help="Сколько стран показывать в каждом топе") args = parser.parse_args() blocks = read_match_blocks(args.xlsx) gerzog = country_stats(blocks, "gerzog_bank") yukon = country_stats(blocks, "yukon_bank") template = Path(args.template).read_text(encoding="utf-8") html_text = template.replace("", build_rows(gerzog, args.limit)) html_text = html_text.replace("", build_rows(yukon, args.limit)) if "" in html_text or "" in html_text: raise SystemExit("В шаблоне не заменились метки строк. Проверь countries_template.html") Path(args.out).write_text(html_text, encoding="utf-8") print(f"Готово: {args.out}") print(f"Обработано матчей: {len(blocks)}") print(f"Gerzog: {len(gerzog)} стран; Yukon: {len(yukon)} стран") subprocess.run(["open", args.out]) #Для + в столбце макс. выигрыш def fmt_max_win(value): value = round(value, 1) if value > 0: prefix = "+" else: prefix = "" if value.is_integer(): return f"{prefix}{int(value)}" return f"{prefix}{str(value).replace('.', ',')}" if __name__ == "__main__": main()
